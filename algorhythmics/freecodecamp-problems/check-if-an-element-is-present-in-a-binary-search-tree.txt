function BinarySearchTree() {
  this.root = null;
  // Only change code below this line
  this.isPresent = function(value) {
    let node = this.root;

    while (node !== null) {
      if (value < node.value) {
        node = node.left;
      } else if (value > node.value) {
        node = node.right;
      } else {
        return true;
      }
    }

    return false;
  }
