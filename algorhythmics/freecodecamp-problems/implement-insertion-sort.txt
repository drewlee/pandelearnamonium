function insertionSort(array) {
  // Only change code below this line
  for (let i = 1; i < array.length; i++) {
    let j = i - 1;
    
    while (j >= 0 && array[j + 1] < array[j]) {
      [array[j], array[j + 1]] = [array[j + 1], array[j]];
      j--;
    }
  }
  
  return array;
  // Only change code above this line
}
