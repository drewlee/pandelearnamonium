function Node(value) {
  this.value = value;
  this.left = null;
  this.right = null;
}
function BinarySearchTree() {
  this.root = null;
  // Only change code below this line
  this.add = function(value) {
    const newNode = new Node(value);

    if (this.root === null) {
      this.root = newNode;
      return;
    }

    let node = this.root;

    while (node !== null) {
      if (value < node.value) {
        if (node.left !== null) {
          node = node.left;
        } else {
          node.left = newNode;
        }
      } else if (value > node.value) {
        if (node.right !== null) {
          node = node.right;
        } else {
          node.right = newNode;
        }
      } else {
        return null;
      }
    }
  }
  // Only change code above this line
}
