function getNumberOfPlants(fieldSize, unit, crop) {
  const conversionRates = {
    acres: 4046.86,
    hectares: 10000,
  };

  const crops = {
    corn: 1,
    wheat: 0.1,
    soybeans: 0.5,
    tomatoes: 0.25,
    lettuce: 0.2,
  };

  const plants = Math.floor(
    fieldSize * conversionRates[unit] / crops[crop]
  );

  return plants;
}
